  body {

    font-family: Arial, sans-serif;

    margin: 20px;

    background-color: #f4f4f4;

  }

  .chart {

    width: 80%;

    margin: 0 auto;

    background-color: white;

    box-shadow: 0 0 10px rgba(0, 0, 0, 0.1);

  }

  .bar {

    fill: #1f77b4;

    transition: fill 0.3s ease;

  }

  .bar:hover {

    fill: #ff7f0e;

  }

  .axis path,

  .axis line {

    fill: none;

<h1 style="text-align: center;">Human Population Growth Over the Last 100 Years</h1>

<div class="chart" id="chart"></div>


<script>

// Sample data for population growth

const data = [

  { year: 1923, population: 1950000000 },

  { year: 1933, population: 2300000000 },

  { year: 1943, population: 2570000000 },

  { year: 1953, population: 2770000000 },

  { year: 1963, population: 3020000000 },

  { year: 1973, population: 3700000000 },

  { year: 1983, population: 4430000000 },

  { year: 1993, population: 5270000000 },

  { year: 2003, population: 6070000000 },

  { year: 2013, population: 7000000000 }

];


// Set up SVG dimensions

const svgWidth = 800;

const svgHeight = 400;

const margin = { top: 40, right: 40, bottom: 60, left: 60 };

const width = svgWidth - margin.left - margin.right;

const height = svgHeight - margin.top - margin.bottom;


// Create SVG element

const svg = d3.select("#chart")

  .append("svg")

  .attr("width", svgWidth)

  .attr("height", svgHeight);


// Create scales

const xScale = d3.scaleBand()

  .domain(data.map(d => d.year))

  .range([margin.left, width - margin.right])

  .padding(0.1);


const yScale = d3.scaleLinear()

  .domain([0, d3.max(data, d => d.population)])

  .nice()

  .range([height - margin.bottom, margin.top]);


// Create bars with animation

svg.selectAll(".bar")

  .data(data)

  .enter().append("rect")

  .attr("class", "bar")

  .attr("x", d => xScale(d.year))

  .attr("y", yScale(0))

  .attr("width", xScale.bandwidth())

  .attr("height", 0)

  .transition()

  .duration(800)

  .delay((d, i) => i * 100)

  .attr("y", d => yScale(d.population))

  .attr("height", d => yScale(0) - yScale(d.population));


// Create axes

const xAxis = d3.axisBottom(xScale).tickSize(0);

const yAxis = d3.axisLeft(yScale).ticks(5).tickSizeOuter(0);


svg.append("g")

  .attr("class", "x-axis")

  .attr("transform", `translate(0, ${height - margin.bottom})`)

  .call(xAxis)

  .selectAll("text")

  .attr("class", "axis-label")

  .attr("dy", "1em")

  .attr("dx", "-1em")

  .attr("transform", "rotate(-45)")

  .style("text-anchor", "end");


svg.append("g")

  .attr("class", "y-axis")

  .attr("transform", `translate(${margin.left}, 0)`)

  .call(yAxis);


// Add axis labels

svg.append("text")

  .attr("class", "axis-label")

  .attr("x", width / 2)

  .attr("y", height + margin.bottom / 1.5)

  .style("text-anchor", "middle")

  .text("Year");


svg.append("text")

  .attr("class", "axis-label")

  .attr("x", -height / 2)

  .attr("y", margin.left / 2)

  .attr("transform", "rotate(-90)")

  .style("text-anchor", "middle")

  .text("Population");


</script>

</body>
